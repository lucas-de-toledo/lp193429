#include <stdio.h>
#include <math.h>

int main (){
    int A, B, C, MaiorAB;
    
    scanf("%d %d %d", &A, &B, &C);
    
    MaiorAB = (A+B+fabs(A-B))/2;
    
    if(C>MaiorAB){
        printf("%d eh o maior\n", C);
    } else {
        printf("%d eh o maior\n", MaiorAB);
    }
    return 0;
}
